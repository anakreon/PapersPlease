priority:
detainment
denial
approval

string interpreter:
'\n' separates items

bulletin:
'Allow [GROUP]'
'Deny [GROUP]'
'[GROUP] (no longer) require [DOCUMENT]'
'Wanted by the State: [NAME]'

[GROUP] => 'foreigners' | 'citizens of [NATION_LIST]' | 'workers' | 'entrants'
[DOCUMENT] - 'access permit' | 'ID card' | 'work pass' | '[ILLNESS] vaccination' | 'passport'
[NATION_LIST] => '[[NATION], [NATION], ...]'
[ILLNESS] => <string>
[NATION] => 'Arstotzka' | 'Antegria' | 'Impor' | 'Kolechia' | 'Obristan' | 'Republia' | 'United Federation'
[SEX] => 'M' | 'F'
[PURPOSE] => 'TRANSIT' | 'WORK'
[NAME] => <string>

documents:
passport: 'ID#: <string>', 'NATION: [NATION]', 'NAME: <string>', 'DOB: <date yyyy.mm.dd>', 'SEX: [SEX]', 'ISS: <string>', 'EXP: <date yyyy.mm.dd>'
access_permit: 'NAME: <string>', 'NATION: [NATION]', 'ID#: <string>', 'PURPOSE: [PURPOSE]', 'DURATION: <string>', 'HEIGHT: <string>', 'WEIGHT: <string>', 'EXP: <date yyyy.mm.dd>'
grant_of_asylum: 'NAME: <string>', 'NATION: [NATION]', 'ID#: <string>', 'DOB: <date yyyy.mm.dd>', 'HEIGHT: <string>', 'EXP: <date yyyy.mm.dd>'

detain -> has conflicting info on documents | is a wanted criminal
deny -> required documents not present (if foreigner & access_permit is required & no access_permit => can use grant_of_asylum, or diplomatic_authorization (must include Arstotzka in list of nations))
    | some documents expired (expiration date < November 22, 1982)
    | required vaccination is missing (certificate_of_vaccination with the required vaccine)
allow -> else

groups:
foreigners => citizens of (Nation != Arstotzka)
citizens of [NATION_LIST] => citizens of (Nation in List)
workers => access_permit - purpose = WORK
entrants => citizens of any Nation

documentParser: documentString -> document
bulletinParser: bulletin -> rules[]
